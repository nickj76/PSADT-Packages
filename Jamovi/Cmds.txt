.\ServiceUI.exe -Process:explorer.exe Deploy-Application.exe -DeploymentType uninstall

Set-RegistryKey -Key HKEY_LOCAL_MACHINE\SOFTWARE\Intune_Installations -Name 'jamovi-2.3.18.0' -Value '"Installed"' -Type String
